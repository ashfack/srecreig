CREATE VIEW Contact
AS 
select `srecreig_base`.`entreprise`.`nomEntreprise` AS `nom Entreprise`,`srecreig_base`.`coordonneespersonne`.`civilite` AS `Civilité`,`srecreig_base`.`coordonneespersonne`.`nom` AS `Nom`,`srecreig_base`.`coordonneespersonne`.`prenom` AS `Prenom`,`srecreig_base`.`coordonneespersonne`.`fonction` AS `Fonction`,`srecreig_base`.`coordonneespersonne`.`telephoneFixe` AS `Téléphone Fixe`,`srecreig_base`.`coordonneespersonne`.`telephoneMobile` AS `Téléphone Mobile`,`srecreig_base`.`coordonneespersonne`.`mail` AS `Mail`,`srecreig_base`.`coordonneespersonne`.`commentaires` AS `Commentaires` from ((`srecreig_base`.`entreprise` join `srecreig_base`.`coordonneespersonne`) join `srecreig_base`.`a_entreprise_coordonneespersonne`) where ((`srecreig_base`.`a_entreprise_coordonneespersonne`.`CoordonneesPersonne_id` = `srecreig_base`.`coordonneespersonne`.`idCoordonneesPersonne`) and (`srecreig_base`.`a_entreprise_coordonneespersonne`.`Entreprise_nomEntreprise` = `srecreig_base`.`a_entreprise_coordonneespersonne`.`Entreprise_nomEntreprise`));
