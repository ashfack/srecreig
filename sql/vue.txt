CREATE VIEW contact 
AS 
SELECT entreprise.nomEntreprise, Coordonneespersonne.civilite, Coordonneespersonne.nom, Coordonneespersonne.prenom, Coordonneespersonne.fonction, Coordonneespersonne.telephoneFixe, Coordonneespersonne.telephoneMobile , Coordonneespersonne.mail, Coordonneespersonne.commentaires FROM entreprise, coordonneespersonne, a_entreprise_coordonneespersonne WHERE a_entreprise_coordonneespersonne.CoordonneesPersonne_id = coordonneespersonne.idCoordonneesPersonne and a_entreprise_coordonneespersonne.Entreprise_nomEntreprise = a_entreprise_coordonneespersonne.Entreprise_nomEntreprise
